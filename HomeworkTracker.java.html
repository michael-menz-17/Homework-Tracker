import java.util.*;

// The HomeworkTask class is designed to represent a single homework task.
class HomeworkTask {
    String title;           // The title of the task (e.g., "Math Homework").
    String dueDate;         // The due date of the task in the format d/m/y (e.g., "25/12/2024").
    String description;     // A brief description of the task (e.g., "Complete chapter 5 exercises").
    String status;          // The current status of the task (e.g., "Pending", "Completed").

    // The constructor method is used to initialize the task when a new HomeworkTask object is created.
    public HomeworkTask(String title, String dueDate, String description) {
        // Assign the provided title to the title variable of the task.
        this.title = title;
        // Assign the provided due date to the dueDate variable of the task.
        this.dueDate = dueDate;
        // Assign the provided description to the description variable of the task.
        this.description = description;
        // Set the status of the task to "Pending" by default.
        this.status = "Pending";
    }

    // This method updates the status of a task to a new value (e.g., "Completed", "In Progress").
    public void updateStatus(String newStatus) {
        // Update the status of the task to the new status provided.
        this.status = newStatus;
    }

    // This method returns a string representation of the task, making it easier to print out the task details.
    public String toString() {
        // Return a string that includes the title, due date, and status of the task.
        return "Title: " + title + ", Due Date: " + dueDate + ", Status: " + status;
    }
}

// The HomeworkTracker class is the main class that manages the collection of tasks and interaction with the user.
public class HomeworkTracker {
    static Scanner scanner = new Scanner(System.in);  // A scanner object to read user input from the console.
    static ArrayList<HomeworkTask> tasks = new ArrayList<>();  // An ArrayList to store the homework tasks.

    // The main method is the entry point of the program where execution begins.
    public static void main(String[] args) {
        boolean running = true; // A boolean variable to control the loop.

        // A loop to keep the program running until the user chooses to exit.
        while (running) {
            // Display the menu to the user so they can choose what they want to do.
            printMenu();
            // Get the user's choice from the menu.
            int choice = getChoice();

            // A series of if-else statements check which option the user selected.
            if (choice == 1) {
                // If the user chooses option 1, call the addNewTask() method to add a new homework task.
                addNewTask();
            } else if (choice == 2) {
                // If the user chooses option 2, call the viewTasks() method to display the existing tasks.
                viewTasks();
            } else if (choice == 3) {
                // If the user chooses option 3, call the updateTaskStatus() method to modify the status of an existing task.
                updateTaskStatus();
            } else if (choice == 4) {
                // If the user chooses option 4, call the removeTask() method to delete a task.
                removeTask();
            } else if (choice == 5) {
                // If the user chooses option 5, display a message that the program is exiting.
                System.out.println("Exiting the program...");
                running = false; // Set running to false to exit the loop.
            } else {
                // If the user enters an invalid option, display this error message.
                System.out.println("Invalid choice! Please try again.");
            }
        }
    }

    // This method prints the available menu options to the console.
    public static void printMenu() {
        // Print the menu header.
        System.out.println("\n--- Homework Tracker ---");
        // Print the first menu option.
        System.out.println("1. Add a new task");
        // Print the second menu option.
        System.out.println("2. View existing tasks");
        // Print the third menu option.
        System.out.println("3. Update task status");
        // Print the fourth menu option.
        System.out.println("4. Remove a task");
        // Print the fifth menu option.
        System.out.println("5. Exit");
    }

    // This method retrieves the user's choice by reading the input from the console.
    public static int getChoice() {
        // Prompt the user to choose an option.
        System.out.print("Choose an option: ");
        // Read the user's choice as an integer.
        int choice = scanner.nextInt();
        // Consume the leftover newline character after reading the integer input.
        scanner.nextLine();
        // Return the user's choice.
        return choice;
    }

    // This method adds a new homework task to the list.
    public static void addNewTask() {
        // Prompt the user to enter the task title.
        System.out.print("Enter task title: ");
        // Read the title as a string input from the user.
        String title = scanner.nextLine();

        // Prompt the user to enter the task due date.
        System.out.print("Enter task due date (d/m/y): ");
        // Read the due date as a string input from the user.
        String dueDate = scanner.nextLine();

        // Prompt the user to enter a short description of the task.
        System.out.print("Enter task description: ");
        // Read the task description from the user.
        String description = scanner.nextLine();

        // Create a new HomeworkTask object using the provided details.
        HomeworkTask task = new HomeworkTask(title, dueDate, description);
        // Add the new task to the tasks list.
        tasks.add(task);
        // Inform the user that the task has been successfully added.
        System.out.println("New task added successfully.");
    }

    // This method displays all the existing tasks in the list.
    public static void viewTasks() {
        // Check if the tasks list is empty.
        if (tasks.isEmpty()) {
            // If no tasks exist, print this message.
            System.out.println("No tasks available.");
            // Exit the method if no tasks are present.
            return;
        }

        // Print a header for the list of tasks.
        System.out.println("\n--- List of Tasks ---");
        // Loop through each task in the tasks list.
        for (int i = 0; i < tasks.size(); i++) {
            // Print each task with its index (i + 1) and formatted details.
            System.out.println((i + 1) + ". " + tasks.get(i).toString());
        }
    }

    // This method allows the user to update the status of a specific task (e.g., marking it as "Completed").
    public static void updateTaskStatus() {
        // Check if the tasks list is empty.
        if (tasks.isEmpty()) {
            // If no tasks exist, print this message.
            System.out.println("No tasks available to update.");
            // Exit the method if no tasks are available to update.
            return;
        }

        // Display the list of tasks so the user can choose one to update.
        viewTasks();

        // Prompt the user to enter the task number to update.
        System.out.print("Enter the task number to update status: ");
        // Read the user's choice as an integer.
        int taskIndex = scanner.nextInt() - 1;
        // Consume the leftover newline character after reading the integer input.
        scanner.nextLine();

        // Check if the taskIndex is valid.
        if (taskIndex < 0 || taskIndex >= tasks.size()) {
            // If the taskIndex is invalid, print an error message.
            System.out.println("Invalid task number.");
            // Exit the method if the taskIndex is invalid.
            return;
        }

        // Prompt the user to enter the new status for the task.
        System.out.print("Enter the new status: ");
        // Read the new status as a string input from the user.
        String newStatus = scanner.nextLine();

        // Update the status of the selected task.
        tasks.get(taskIndex).updateStatus(newStatus);
        // Inform the user that the task status has been successfully updated.
        System.out.println("Task status updated successfully.");
    }

    // This method allows the user to remove a specific task from the list.
    public static void removeTask() {
        // Check if the tasks list is empty.
        if (tasks.isEmpty()) {
            // If no tasks exist, print this message.
            System.out.println("No tasks available to remove.");
            // Exit the method if no tasks are available to remove.
            return;
        }

        // Display the list of tasks so the user can choose one to remove.
        viewTasks();

        // Prompt the user to enter the task number to remove.
        System.out.print("Enter the task number to remove: ");
        // Read the user's choice as an integer.
        int taskIndex = scanner.nextInt() - 1;
        // Consume the leftover newline character after reading the integer input.
        scanner.nextLine();

        // Check if the taskIndex is valid.
        if (taskIndex < 0 || taskIndex >= tasks.size()) {
            // If the taskIndex is invalid, print an error message.
            System.out.println("Invalid task number.");
            // Exit the method if the taskIndex is invalid.
            return;
        }

        // Remove the selected task from the list.
        tasks.remove(taskIndex);
        // Inform the user that the task has been successfully removed.
        System.out.println("Task removed successfully.");
    }
}
